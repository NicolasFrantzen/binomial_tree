11pub const MAX_TREE_SIZE: usize = 128;
12const PRE_ALLOCATED_STACK: &[&[NodeName2]] = binomial_tree_stack!(128);
13pub(crate) const MAX_CAPACITY: usize = calculate_capacity(MAX_TREE_SIZE);

15#[derive(Debug, Default)]

20impl StaticBinomialTreeMap {
21    pub fn with_capacity(capacity: usize) -> StaticBinomialTreeMap {

24        StaticBinomialTreeMap { stack }
25    }

28#[derive(Debug)]
29pub struct StaticContainer {
30    arena: Arena<BinomTreeValueType>,
31    pub(crate) map: HashMap<NodeName2, Id<BinomTreeValueType>>,
32}
33
34impl Default for StaticContainer {
35    fn default() -> Self {
36        Self {
37            arena: Arena::with_capacity(MAX_CAPACITY),
38            map: HashMap::<NodeName2, Id<BinomTreeValueType>>::with_capacity(MAX_CAPACITY),
39        }
40    }
41}

43impl BinomialTreeMapImpl for StaticContainer {
44    type NodeNameType = NodeName2;
45    type NumericType = BinomialTreeMapNumericType;
46    type ValueType = BinomTreeValueType;
47
48    fn get(&self, node_name: &Self::NodeNameType) -> Option<&Self::ValueType> {
49        let id = self.map.get(node_name)?;
50        Some(&self.arena[*id])
51    }
52
53    fn get_next_step(&self, node_name: &Self::NodeNameType) -> Option<&Self::ValueType> {
54        let (key, _) = self.map.get_key_value(&node_name.up()).unwrap();
55        self.get(&key.down())
56    }
57
58    fn set(&mut self, node_name: &Self::NodeNameType, value: Self::NumericType) {
59        let id = self.arena.alloc(OnceCell::new());
60        self.map.entry(node_name.clone()).or_insert(id);
61
62        self.arena[id].set(value).unwrap();
63    }
64}
